import json, time, os, csv
from datetime import datetime, timedelta

SERAGONE_DIR = '/home/ubuntu/seragone'
OUTPUT = '/var/www/html/bulletin_mobile.json'
VALIDATION_START = '2026-04-14'
VALIDATION_DAYS = 60

def read_json(path):
    with open(path) as f:
        return json.load(f)

def count_signals():
    csv_path = os.path.join(SERAGONE_DIR, 'modules_historique_complet.csv')
    start = datetime.strptime(VALIDATION_START, '%Y-%m-%d')
    signaux = []
    try:
        with open(csv_path) as f:
            reader = csv.DictReader(f)
            rows = list(reader)
        for i, row in enumerate(rows):
            date_str = row.get('date', '').strip()
            if not date_str:
                continue
            try:
                date = datetime.strptime(date_str, '%Y-%m-%d')
            except:
                continue
            if date < start:
                continue
            dist = float(row.get('dist_M0', 0))
            shan = float(row.get('shannon', 0))
            prix = float(row.get('prix', 0))
            if dist > 0.25 and shan < 0.25:
                resultat = 'EN ATTENTE'
                gain = 0
                if i + 5 < len(rows):
                    prix_apres = float(rows[i+5].get('prix', 0))
                    if prix_apres > 0 and prix > 0:
                        gain = round((prix_apres - prix) / prix * 100, 2)
                        resultat = 'OUI' if gain > 0 else 'NON'
                signaux.append({
                    'date': date_str,
                    'prix': prix,
                    'distance': dist,
                    'shannon': shan,
                    'resultat': resultat,
                    'gain_pct': gain
                })
    except Exception as e:
        print(f"Erreur lecture CSV: {e}")
    return signaux

def generate():
    state = read_json(os.path.join(SERAGONE_DIR, 'state.json'))
    brisance = read_json(os.path.join(SERAGONE_DIR, 'brisance_state.json'))
    qjour = read_json(os.path.join(SERAGONE_DIR, 'q_jour_state.json'))
    paralleles = read_json(os.path.join(SERAGONE_DIR, 'mondes_paralleles_state.json'))

    now = datetime.now()
    start = datetime.strptime(VALIDATION_START, '%Y-%m-%d')
    jours = min(60, max(1, (now - start).days + 1))

    signaux = count_signals()
    signaux_ok = sum(1 for s in signaux if s['resultat'] == 'OUI')
    signaux_rate = sum(1 for s in signaux if s['resultat'] == 'NON')
    signaux_attente = sum(1 for s in signaux if s['resultat'] == 'EN ATTENTE')

    # Score precedent - NE DESCEND JAMAIS
    prev_score = 0
    try:
        with open(OUTPUT) as f:
            prev = json.load(f)
        prev_score = prev.get('score_seragone', 0)
    except:
        pass

    # Score : chaque signal juste = +5 points definitifs (20 justes = 100%)
    # Maturite = plafond temporel (jour/60 * 100)
    # Score = min(acquis, maturite) puis max(score, precedent)
    points_acquis = signaux_ok * 5
    maturite = min(100, round(jours / 60 * 100, 1))
    score_calcule = min(points_acquis, maturite)
    score = max(score_calcule, prev_score)

    if jours < 15:
        confiance = 'TROP TOT'
    elif score < 30:
        confiance = 'INSUFFISANT'
    elif score < 55:
        confiance = 'MITIGE'
    elif score < 80:
        confiance = 'PROMETTEUR'
    else:
        confiance = 'VALIDE'

    prix = state.get('prix', 0)
    recursifs_actifs = state.get('n_mondes_actifs', 0)
    paralleles_actifs = paralleles.get('n_actif', 0)
    paralleles_total = paralleles.get('n_mondes', 305)
    convergence_pct = paralleles.get('convergence_pct', 0)
    bris = brisance.get('etat', 'RAS')

    try:
        with open(os.path.join(SERAGONE_DIR, 'modules_historique_complet.csv')) as f:
            lines = f.readlines()
            last = lines[-1].strip().split(',')
            shannon = float(last[16]) if len(last) > 16 else 0
            distance = float(last[7]) if len(last) > 7 else 0
    except:
        shannon = 0
        distance = 0

    geste = qjour.get('m0_geste', 'MAINTENIR')

    tore_bear = geste in ['RETRECIR', 'ATTENDRE']
    if tore_bear:
        meteo = 'orageux'
    elif distance > 0.30 and shannon < 0.15:
        meteo = 'arc-en-ciel'
    elif shannon < 0.15:
        meteo = 'soleil'
    elif shannon < 0.30:
        meteo = 'nuageux'
    elif shannon < 0.50:
        meteo = 'pluie'
    else:
        meteo = 'tempete'

    date_str = now.strftime('%d %B').lstrip('0')
    if geste == 'OUVRIR':
        journal = f"{date_str} - Signal clair. Distance {distance:.3f}, brouillard faible ({shannon:.3f}). J'y vais."
    elif geste == 'MAINTENIR':
        journal = f"{date_str} - Je vois de la tension mais le bruit est encore present (shannon {shannon:.3f}). J'attends mon moment."
    elif geste == 'RETRECIR':
        journal = f"{date_str} - Le vent tourne. Je reduis mon exposition. Prudence."
    elif geste == 'ATTENDRE':
        journal = f"{date_str} - Trop de brouillard. Je ne vois rien. J'attends."
    else:
        journal = f"{date_str} - Le village observe. Rien de special aujourd'hui."

    bulletin = {
        'timestamp': now.isoformat(),
        'prix_btc': prix,
        'score_seragone': score,
        'confiance': confiance,
        'jours_ecoules': jours,
        'jours_objectif': VALIDATION_DAYS,
        'signaux_total': len(signaux),
        'signaux_ok': signaux_ok,
        'signaux_rate': signaux_rate,
        'signaux_attente': signaux_attente,
        'derniers_signaux': signaux[-5:] if signaux else [],
        'convergence': {
            'score': convergence_pct,
            'mondes_actifs': paralleles_actifs,
            'mondes_total': paralleles_total,
            'recursifs_actifs': recursifs_actifs
        },
        'couverture': {
            'pourcentage': 74.0,
            'mondes_total': 845
        },
        'protection': {
            'score': 100 if bris == 'RAS' else 50,
            'brisance': bris,
            'gardien': 'OUI'
        },
        'systeme': {
            'distance': distance,
            'shannon': shannon,
            'geste': geste,
            'mode': 'SIGNAL' if geste == 'OUVRIR' else 'WAIT'
        },
        'meteo': meteo,
        'journal': journal,
        'processus': {
            'collector': True,
            'brain': True,
            'brisance': True
        }
    }

    with open(OUTPUT, 'w') as f:
        json.dump(bulletin, f, indent=2, ensure_ascii=False)

    print(f"[{now.strftime('%H:%M:%S')}] OK score:{score}% (acquis:{points_acquis} mat:{maturite} prev:{prev_score}) jour:{jours}/60 sig:{signaux_ok}/{len(signaux)}")

if __name__ == '__main__':
    print(f"Bulletin mobile - validation depuis {VALIDATION_START}")
    while True:
        try:
            generate()
        except Exception as e:
            print(f"Erreur: {e}")
        time.sleep(300)
